struct node{
int data; 
struct node *next;
}typedef node *NODE;
NODE createNode(){
NODE temp;
temp=malloc(sizeof(Struct node));
temp->next=NULL;
return temp;
}
NODE insertAtBegin(NODE first, int x)
{
NODE temp=createNode();
temp->data=x;
if(first==NULL)
{
temp->next=first;
first=temp;

}
NODE display(NODE first)
{
NODE temp=first;
while(temp!=NULL)
{
printf("%d--> ",temp->data);
temp=temp->next;

}
return first;
}
NODE count(NODE first)
{
NODE temp=first;
int count=0;
while(temp!=NULL)
{
count++;
temp=temp->next;
}
return count;
}
